FormVenda.jsx
import React, { useState } from "react";
import { Form, Button } from "semantic-ui-react";
import axios from "axios";
import InputMask from "react-input-mask";

export default function FormVenda(){

    const [venda, setVenda] = useState({
        cliente: "",
        produto: "",
        statusVenda: "",
        dataVenda: "",
        valorTotal: "",
        observacao: "",
        retiradaEmLoja: true
    });

    function handleChange(e, {name, value}){
        setVenda({...venda, [name]: value});
    }

    function salvar(){
        axios.post("http://localhost:8080/api/venda", venda)
        .then(() => alert("Venda cadastrada com sucesso"))
        .catch(() => alert("Erro ao salvar"));
    }

    return(
        <Form>

            <Form.Input
                label="Cliente"
                name="cliente"
                value={venda.cliente}
                onChange={handleChange}
            />

            <Form.Input
                label="Produto"
                name="produto"
                value={venda.produto}
                onChange={handleChange}
            />

            <Form.Select
                label="Status"
                name="statusVenda"
                options={[
                    { key:1, text:'Pedido Cancelado', value:'Pedido Cancelado'},
                    { key:2, text:'Aguardando Pagamento', value:'Aguardando Pagamento'},
                    { key:3, text:'Pago', value:'Pago'},
                    { key:4, text:'Entregue', value:'Entregue'}
                ]}
                onChange={handleChange}
            />

            <Form.Field>
                <label>Data Venda</label>
                <InputMask
                    mask="99/99/9999"
                    name="dataVenda"
                    value={venda.dataVenda}
                    onChange={(e)=> setVenda({...venda, dataVenda:e.target.value})}
                />
            </Form.Field>

            <Form.Input
                label="Valor Total"
                name="valorTotal"
                value={venda.valorTotal}
                onChange={handleChange}
            />

            <Form.TextArea
                label="Observação"
                name="observacao"
                value={venda.observacao}
                onChange={handleChange}
            />

            <Form.Group>
                <label>Retirada em Loja</label>

                <Form.Radio
                    label="Sim"
                    name="retiradaEmLoja"
                    value={true}
                    checked={venda.retiradaEmLoja === true}
                    onChange={()=> setVenda({...venda, retiradaEmLoja:true})}
                />

                <Form.Radio
                    label="Não"
                    name="retiradaEmLoja"
                    value={false}
                    checked={venda.retiradaEmLoja === false}
                    onChange={()=> setVenda({...venda, retiradaEmLoja:false})}
                />
            </Form.Group>

            <Button primary onClick={salvar}>
                Salvar
            </Button>

        </Form>
    );
}

ListVenda.jsx
import React, {useEffect, useState} from "react";
import axios from "axios";
import { Table, Button, Modal } from "semantic-ui-react";

export default function ListVenda(){

    const [lista, setLista] = useState([]);
    const [open, setOpen] = useState(false);
    const [obs, setObs] = useState("");

    useEffect(()=>{
        carregar();
    },[]);

    function carregar(){
        axios.get("http://localhost:8080/api/venda")
        .then(response => setLista(response.data));
    }

    function remover(id){
        axios.delete("http://localhost:8080/api/venda/"+id)
        .then(()=> carregar());
    }

    function mostrarObs(texto){
        setObs(texto);
        setOpen(true);
    }

    return(
        <>

        <Table celled>
            <Table.Header>
                <Table.Row>
                    <Table.HeaderCell>Cliente</Table.HeaderCell>
                    <Table.HeaderCell>Produto</Table.HeaderCell>
                    <Table.HeaderCell>Status</Table.HeaderCell>
                    <Table.HeaderCell>Data</Table.HeaderCell>
                    <Table.HeaderCell>Valor</Table.HeaderCell>
                    <Table.HeaderCell>Retirada</Table.HeaderCell>
                    <Table.HeaderCell>Ações</Table.HeaderCell>
                </Table.Row>
            </Table.Header>

            <Table.Body>
                {lista.map(v => (
                    <Table.Row key={v.id}>
                        <Table.Cell>{v.cliente}</Table.Cell>
                        <Table.Cell>{v.produto}</Table.Cell>
                        <Table.Cell>{v.statusVenda}</Table.Cell>
                        <Table.Cell>{v.dataVenda}</Table.Cell>
                        <Table.Cell>{v.valorTotal}</Table.Cell>
                        <Table.Cell>{v.retiradaEmLoja ? "Sim" : "Não"}</Table.Cell>
                        <Table.Cell>

                            <Button color="red" onClick={()=> remover(v.id)}>
                                Remover
                            </Button>

                            <Button color="blue" onClick={()=> mostrarObs(v.observacao)}>
                                Observação
                            </Button>

                        </Table.Cell>
                    </Table.Row>
                ))}
            </Table.Body>
        </Table>

        <Modal open={open} onClose={()=> setOpen(false)}>
            <Modal.Header>Observação</Modal.Header>
            <Modal.Content>
                {obs}
            </Modal.Content>
            <Modal.Actions>
                <Button onClick={()=> setOpen(false)}>Fechar</Button>
            </Modal.Actions>
        </Modal>

        </>
    );
}

Uma API RESTful é um serviço na web que disponibiliza informações ou funcionalidades através de 
endereços (URLs). Ela utiliza os métodos padrão do protocolo HTTP para acessar e modificar esses
 recursos, não guarda informações da sessão entre pedidos feitos pelo cliente e segue regras 
 git UsernamePasswordAuthenticationFilterabertas para garantir que diferentes sistemas possam se comunicar facilmente.
